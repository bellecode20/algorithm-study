# 카페 운영 


```python

def solution(menu, order, k):
    
    # 들어온 시점 기준으로 내보낸 다음 계산을 하면 된다.
    # 1. 새 손님 입장
    # 2. 최대한 내보내기
    # 3. 사람 수 카운트 
    
    메뉴 = menu
    주문 = order
    간격 = k
    정답 = 0
    현재시간 = 0
    스택 = [0]
    스택_인덱스 = 1 # 새로 들어올 인덱스 
    for 입장번호 in range(len(주문)):
        입장시간 = 입장번호*간격
        퇴장시간 = max(스택[-1],입장시간)+메뉴[주문[입장번호]]
        스택.append(퇴장시간)

        while 스택[스택_인덱스] <= 입장시간:
            스택_인덱스 += 1
        정답 = max(정답, len(스택)-스택_인덱스)

    return 정답
    ```
